## 3. 数据库设计

### 3.1 核心数据表

#### users (用户表)
```sql
CREATE TABLE users (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  email TEXT UNIQUE NOT NULL,
  username TEXT UNIQUE,
  avatar_url TEXT,
  display_name TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);
```

#### timelines (时间线表)
```sql
CREATE TABLE timelines (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  owner_id UUID REFERENCES users(id) ON DELETE CASCADE,
  title TEXT NOT NULL,
  description TEXT,
  is_public BOOLEAN DEFAULT FALSE,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX idx_timelines_owner ON timelines(owner_id);
```

#### timeline_members (时间线成员表)
```sql
CREATE TABLE timeline_members (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  timeline_id UUID REFERENCES timelines(id) ON DELETE CASCADE,
  user_id UUID REFERENCES users(id) ON DELETE CASCADE,
  role TEXT DEFAULT 'member', -- 'owner', 'admin', 'member'
  joined_at TIMESTAMPTZ DEFAULT NOW(),
  UNIQUE(timeline_id, user_id)
);

CREATE INDEX idx_members_timeline ON timeline_members(timeline_id);
CREATE INDEX idx_members_user ON timeline_members(user_id);
```

#### events (事件表)
```sql
CREATE TABLE events (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  timeline_id UUID REFERENCES timelines(id) ON DELETE CASCADE,
  creator_id UUID REFERENCES users(id),
  title TEXT NOT NULL,
  description TEXT,
  start_time TIMESTAMPTZ NOT NULL,
  end_time TIMESTAMPTZ NOT NULL,
  location TEXT,
  is_all_day BOOLEAN DEFAULT FALSE,
  color TEXT DEFAULT '#6366f1',
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX idx_events_timeline ON events(timeline_id);
CREATE INDEX idx_events_time ON events(start_time, end_time);
```

#### event_attendees (事件参与者表)
```sql
CREATE TABLE event_attendees (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  event_id UUID REFERENCES events(id) ON DELETE CASCADE,
  user_id UUID REFERENCES users(id),
  status TEXT DEFAULT 'pending', -- 'pending', 'accepted', 'declined'
  UNIQUE(event_id, user_id)
);

CREATE INDEX idx_attendees_event ON event_attendees(event_id);
```

#### invitations (邀请表)
```sql
CREATE TABLE invitations (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  timeline_id UUID REFERENCES timelines(id) ON DELETE CASCADE,
  invite_token TEXT UNIQUE NOT NULL,
  created_by UUID REFERENCES users(id),
  expires_at TIMESTAMPTZ,
  max_uses INTEGER DEFAULT 1,
  used_count INTEGER DEFAULT 0
);

CREATE INDEX idx_invitations_token ON invitations(invite_token);
```

### 3.2 Row Level Security (RLS) 策略

```sql
-- 启用RLS
ALTER TABLE users ENABLE ROW LEVEL SECURITY;
ALTER TABLE timelines ENABLE ROW LEVEL SECURITY;
ALTER TABLE events ENABLE ROW LEVEL SECURITY;
ALTER TABLE timeline_members ENABLE ROW LEVEL SECURITY;

-- 用户只能查看/编辑自己的信息
CREATE POLICY "Users can view own profile"
  ON users FOR SELECT USING (auth.uid() = id);

CREATE POLICY "Users can update own profile"
  ON users FOR UPDATE USING (auth.uid() = id);

-- 时间线成员可以查看所属时间线
CREATE POLICY "Members can view timeline"
  ON timelines FOR SELECT
  USING (id IN (
    SELECT timeline_id FROM timeline_members WHERE user_id = auth.uid()
  ));

CREATE POLICY "Members can view timeline events"
  ON events FOR SELECT
  USING (timeline_id IN (
    SELECT timeline_id FROM timeline_members WHERE user_id = auth.uid()
  ));

-- 时间线成员可以创建事件
CREATE POLICY "Members can create events"
  ON events FOR INSERT
  WITH CHECK (timeline_id IN (
    SELECT timeline_id FROM timeline_members WHERE user_id = auth.uid()
  ));
```

## 5. 状态管理架构

### 5.1 Riverpod Provider架构

```dart
// providers/auth_provider.dart
@riverpod
class Auth extends _$Auth {
  @override
  AuthState build() {
    return const AuthState.initial();
  }

  Future<void> signIn(String email, String password) async {
    state = const AuthState.loading();
    final result = await ref.read(authRepositoryProvider).signIn(email, password);
    state = result.fold(
      (error) => AuthState.error(error),
      (user) => AuthState.authenticated(user),
    );
  }
}

// providers/timeline_provider.dart
@riverpod
class Timeline extends _$Timeline {
  @override
  FutureOr<List<Timeline>> build() async {
    return ref.read(timelineRepositoryProvider).getTimelines();
  }

  Future<void> createTimeline(CreateTimelineDto dto) async {
    final timeline = await ref.read(timelineRepositoryProvider).create(dto);
    state = AsyncValue.data([...state.value ?? [], timeline]);
  }
}
```

### 5.2 实时同步策略

```dart
// services/realtime_service.dart
class RealtimeService {
  final SupabaseClient _client;
  final Ref _ref;

  void subscribeToTimelineEvents(String timelineId) {
    _client.channel('timeline_events').on(
      RealtimeListenEventType.postgres_changes,
      ChannelFilter(
        event: '*',
        schema: 'public',
        table: 'events',
        filter: 'timeline_id=eq.$timelineId',
      ),
      (payload, [ref]) {
        final event = Event.fromJson(payload.newRecord);
        _ref.read(timelineEventsProvider.notifier).addEvent(event);
      },
    ).subscribe();
  }
}
```

## 6. AI集成架构

### 6.1 AI服务接口

```dart
class AIService {
  final String apiKey;
  final String baseUrl;

  Future<ScheduleParseResult> parseSchedule(String input) async {
    final prompt = _buildParsePrompt(input);
    final response = await _callLLM(prompt);
    return ScheduleParseResult.fromJson(response);
  }

  String _buildParsePrompt(String userInput) {
    return '''
你是一个智能日程助手。请将用户的自然语言输入解析为结构化的日程事件。

用户输入: $userInput

请以JSON格式返回，包含以下字段:
- title: 事件标题
- startTime: 开始时间 (ISO8601格式)
- endTime: 结束时间 (ISO8601格式)
- location: 地点 (可选)
- description: 描述 (可选)

返回JSON:
''';
  }
}
```
