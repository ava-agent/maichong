    });

    group('validation', () {
      test('should fail when title is empty', () {
        expect(
          () => Event(
            id: '1',
            timelineId: 'timeline-1',

// test/unit/data/repositories/event_repository_test.dart

import 'package:flutter_test/flutter_test.dart';
import 'package:mocktail/mocktail.dart';
import 'package:maichong/data/repositories/event_repository.dart';
import 'package:maichong/data/datasources/event_datasource.dart';

class MockEventDataSource extends Mock implements EventDataSource {}

void main() {
  late EventRepository repository;
  late MockEventDataSource mockDataSource;

  setUp(() {
    mockDataSource = MockEventDataSource();
    repository = EventRepository(mockDataSource);
  });

  group('getEvents', () {
    final timelineId = 'timeline-1';
    final events = [
      Event(
        id: '1',
        timelineId: timelineId,
        title: 'Event 1',
        startTime: DateTime(2026, 2, 10, 14, 0),
        endTime: DateTime(2026, 2, 10, 15, 0),
      ),
    ];

    test('should return events when datasource call is successful', () async {
      // Arrange
      when(() => mockDataSource.getEvents(timelineId))
          .thenAnswer((_) async => events);

      // Act
      final result = await repository.getEvents(timelineId);

      // Assert
      expect(result, equals(events));
      verify(() => mockDataSource.getEvents(timelineId)).called(1);
    });

    test('should cache events after successful fetch', () async {
      // Arrange
      when(() => mockDataSource.getEvents(timelineId))
          .thenAnswer((_) async => events);

      // Act
      await repository.getEvents(timelineId);
      final cached = await repository.getEvents(timelineId);

      // Assert
      expect(cached, equals(events));
      verify(() => mockDataSource.getEvents(timelineId)).called(1);
    });

    test('should throw exception when datasource call fails', () async {
      // Arrange
      when(() => mockDataSource.getEvents(timelineId))
          .thenThrow(NetworkException());

      // Act & Assert
      expect(
        () => repository.getEvents(timelineId),
        throwsA(isA<NetworkException>()),
      );
    });
  });

  group('createEvent', () {
    final createDto = CreateEventDto(
      timelineId: 'timeline-1',
      title: 'New Event',
      startTime: DateTime(2026, 2, 10, 14, 0),
      endTime: DateTime(2026, 2, 10, 15, 0),
    );

    test('should create and return event', () async {
      // Arrange
      final createdEvent = Event(
        id: '1',
        timelineId: createDto.timelineId,
        title: createDto.title,
        startTime: createDto.startTime,
        endTime: createDto.endTime,
      );
      when(() => mockDataSource.createEvent(createDto))
          .thenAnswer((_) async => createdEvent);

      // Act
      final result = await repository.createEvent(createDto);

      // Assert
      expect(result, equals(createdEvent));
      verify(() => mockDataSource.createEvent(createDto)).called(1);
    });

    test('should invalidate cache after creating event', () async {
      // Arrange
      when(() => mockDataSource.createEvent(createDto))
          .thenAnswer((_) async => Event(
                id: '1',
                timelineId: createDto.timelineId,
                title: createDto.title,
                startTime: createDto.startTime,
                endTime: createDto.endTime,
              ));

      // Act
      await repository.createEvent(createDto);

      // Assert - next fetch should call datasource
      await repository.getEvents(createDto.timelineId);
      verify(() => mockDataSource.getEvents(createDto.timelineId)).called(1);
    });
  });
}

// test/unit/core/utils/date_utils_test.dart

import 'package:flutter_test/flutter_test.dart';
import 'package:maichong/core/utils/date_utils.dart';

void main() {
  group('DateUtils', () {
    group('isToday', () {
      test('should return true for today', () {
        final today = DateTime.now();
        expect(DateUtils.isToday(today), true);
      });

      test('should return false for yesterday', () {
        final yesterday = DateTime.now().subtract(const Duration(days: 1));
        expect(DateUtils.isToday(yesterday), false);
      });

      test('should return false for tomorrow', () {
        final tomorrow = DateTime.now().add(const Duration(days: 1));
        expect(DateUtils.isToday(tomorrow), false);
      });
    });

    group('isSameDay', () {
      test('should return true for same day different times', () {
        final date1 = DateTime(2026, 2, 10, 14, 0);
        final date2 = DateTime(2026, 2, 10, 18, 30);
        expect(DateUtils.isSameDay(date1, date2), true);
      });

      test('should return false for different days', () {
        final date1 = DateTime(2026, 2, 10);
        final date2 = DateTime(2026, 2, 11);
        expect(DateUtils.isSameDay(date1, date2), false);
      });
    });

    group('parseNaturalTime', () {
      test('should parse "明天"', () {
        final result = DateUtils.parseNaturalTime('明天');
        final expected = DateTime.now().add(const Duration(days: 1));
        expect(DateUtils.isSameDay(result, expected), true);
      });

      test('should parse "下午3点"', () {
        final now = DateTime(2026, 2, 10, 10, 0);
        final result = DateUtils.parseNaturalTime('下午3点', reference: now);
        expect(result.hour, 15);
        expect(result.minute, 0);
      });

      test('should handle "下周五"', () {
        final result = DateUtils.parseNaturalTime('下周五');
        expect(result.weekday, DateTime.friday);
        expect(result.isAfter(DateTime.now()), true);
      });
    });
  });
}

        '明天下午3点和小明喝咖啡',
      );
      await tester.pumpAndSettle();

      // 4. 发送消息
      await tester.tap(find.byIcon(Icons.send));
      await tester.pumpAndSettle();

      // 5. 等待AI响应
      await tester.pump(const Duration(seconds: 3));

      // 6. 验证解析结果预览
      expect(find.text('明天 15:00'), findsOneWidget);
      expect(find.text('和小明喝咖啡'), findsOneWidget);

      // 7. 确认创建
      await tester.tap(find.text('确认创建'));
      await tester.pumpAndSettle();

      // 8. 验证事件已创建
      expect(find.text('和小明喝咖啡'), findsOneWidget);
    });
  });
}

## CI/CD集成

### GitHub Actions配置

```yaml
# .github/workflows/test.yml

name: Tests

on:
  push:
    branches: [ main, develop ]
  pull_request:
    branches: [ main, develop ]

jobs:
  test:
    runs-on: ubuntu-latest

    steps:
      - uses: actions/checkout@v3

      - name: Setup Flutter
        uses: subosito/flutter-action@v2
        with:
          flutter-version: '3.24.0'
          cache: true

      - name: Install dependencies
        run: flutter pub get

      - name: Generate code
        run: flutter pub run build_runner build --delete-conflicting-outputs

      - name: Run tests
        run: flutter test --coverage

      - name: Upload coverage
        uses: codecov/codecov-action@v3
        with:
          files: coverage/lcov.info

      - name: Check coverage threshold
        run: |
          coverage=$(lcov --summary coverage/lcov.info | grep lines | awk '{print $2}' | sed 's/%//')
          if (( $(echo "$coverage < 80" | bc -l) )); then
            echo "Coverage $coverage% is below 80%"
            exit 1
          fi
```
